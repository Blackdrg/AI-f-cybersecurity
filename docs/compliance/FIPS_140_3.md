## Implementation Details

### FIPS Mode Configuration

```python
# backend/app/security/fips_config.py
import os
from cryptography.hazmat.backends import default_backend

FIPS_ENABLED = os.getenv("FIPS_MODE", "false").lower() == "true"

def get_fips_backend():
    """Get FIPS-compliant cryptography backend."""
    if FIPS_ENABLED:
        # Enforce FIPS-approved algorithms only
        from cryptography.hazmat.primitives import hashes
        from cryptography.hazmat.primitives.ciphers import Cipher, algorithms, modes
        
        # Only allow FIPS-approved primitives
        ALLOWED_ALGORITHMS = {
            "hashes": ["SHA256", "SHA384", "SHA512"],
            "ciphers": ["AES256-GCM", "AES256-CBC"],
            "keywrap": ["AES-KW"],
            "kdf": ["HKDF-SHA256", "PBKDF2-SHA256"]
        }
        
        return "fips", ALLOWED_ALGORITHMS
    
    return "default", None

def validate_fips_compliance(algorithm: str, operation: str) -> bool:
    """Validate algorithm is FIPS-approved."""
    backend, allowed = get_fips_backend()
    if backend == "default":
        return True  # FIPS mode not enabled
    
    return algorithm in allowed.get(operation, [])
```

### Key Management FIPS Compliance

```python
# backend/app/security/key_management.py
import os
from typing import Tuple

class FIPSKeyManager:
    """FIPS 140-3 compliant key management."""
    
    def __init__(self):
        self.fips_mode = os.getenv("FIPS_MODE", "false").lower() == "true"
    
    def generate_aes_key(self, length: int = 256) -> bytes:
        """Generate FIPS-approved AES key."""
        if self.fips_mode and length not in [128, 192, 256]:
            raise ValueError("Key length must be 128, 192, or 256 bits in FIPS mode")
        
        # Use FIPS-approved RNG (os.urandom is NIST SP 800-90B compliant)
        return os.urandom(length // 8)
    
    def derive_key(self, password: bytes, salt: bytes, iterations: int = 100000) -> bytes:
        """Derive key using FIPS-approved PBKDF2."""
        from cryptography.hazmat.primitives.kdf.pbkdf2 import PBKDF2HMAC
        from cryptography.hazmat.primitives import hashes
        
        if self.fips_mode and iterations < 10000:
            raise ValueError("Minimum 10000 iterations required in FIPS mode")
        
        kdf = PBKDF2HMAC(
            algorithm=hashes.SHA256(),
            length=32,
            salt=salt,
            iterations=iterations,
        )
        return kdf.derive(password)
```

### Allowed Algorithms in FIPS Mode
| Type | Algorithm | Minimum Key Size |
|------|-----------|------------------|
| Symmetric | AES | 128 bits |
| Hash | SHA-2 family | 256 bits |
| RSA | RSA | 2048 bits |
| ECC | ECDSA | P-256 curve |
| KDF | PBKDF2, HKDF | SHA-256 |

### Prohibited Algorithms
- MD5 (except HMAC)
- SHA-1 (except HMAC)
- RC4
- DES, 3DES
- RSA keys < 2048 bits
- ECC < P-256
